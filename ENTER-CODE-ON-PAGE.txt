   ✅ أدخل الكود في الصفحة - Enter Code on Page
═══════════════════════════════════════════════════════

✅ أنت الآن في صفحة "Authorize Device"!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 1: ابحث عن الكود في Terminal

   في Terminal، ابحث عن:
   user_code=DVZT-XXXT
   
   → الكود هو الجزء بعد "user_code="
   → في المثال أعلاه: DVZT-XXXT
   → (الكود الحقيقي قد يكون مختلفاً)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 2: انسخ الكود من Terminal

   → انسخ الكود (4 أحرف + - + 4 أحرف)
   → مثال: DVZT-XXXT
   → أو: ABCD-EFGH

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 3: أدخل الكود في الصفحة

   → في صفحة "Authorize Device"
   → أدخل أول 4 أحرف في الحقول الأربع الأولى
   → ثم أدخل آخر 4 أحرف في الحقول الثلاث الأخيرة
   
   مثال: DVZT-XXXT
   → الحقل 1: D
   → الحقل 2: V
   → الحقل 3: Z
   → الحقل 4: T
   → (بعد الشرطة)
   → الحقل 5: X
   → الحقل 6: X
   → الحقل 7: X
   → الحقل 8: T

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 4: اضغط "Allow"

   → بعد إدخال الكود
   → اضغط الزر "Allow" (أسفل الصفحة)
   → ✅ سيرجع Terminal تلقائياً!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

💡 نصيحة:

   → الكود يظهر في Terminal كـ:
      user_code=XXXX-XXXX
   
   → انسخ فقط الجزء بعد "=" (XXXX-XXXX)
   → أدخله في الصفحة

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
