   Terminal لا يتعرف على "vercel"

✅ الحل: استخدم npx (لا يحتاج تثبيت!)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 في Terminal، انسخ والصق:

1️⃣ سجّل دخول:

   npx vercel login

   → سيظهر رابط في Terminal
   → انسخ الرابط وافتحه في المتصفح
   → اضغط "Authorize"
   → ✅ سيرجع تلقائياً

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

2️⃣ انشر المشروع:

   npx vercel

   → سيطلب إعدادات:
   → اضغط Enter للقيم الافتراضية
   → أضف Environment Variables

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

💡 npx يعمل مباشرة بدون تثبيت!
