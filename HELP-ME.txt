   ❓ كيف أساعدك؟
═══════════════════════════════════════════════════════

إذا كنت تواجه مشكلة في أي خطوة، أخبرني:

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📍 1. في أي صفحة أنت الآن؟

   □ صفحة تسجيل الدخول (Sign Up/Log In)
   □ Dashboard الرئيسي
   □ صفحة "New Project" أو "Add New"
   □ صفحة رفع الملفات
   □ صفحة إعداد المشروع
   □ صفحة Environment Variables
   □ صفحة أخرى (اذكرها)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

👀 2. ماذا ترى على الشاشة؟

   اذكر:
   - الأزرار الموجودة
   - النصوص المكتوبة
   - أي رسائل خطأ (إن وجدت)
   - ما الذي يمنعك من المتابعة؟

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🛑 3. في أي خطوة توقفت؟

   □ لا أستطيع تسجيل الدخول
   □ لا أجد زر "New Project"
   □ لا أستطيع رفع المجلد
   □ لا أجد "Environment Variables"
   □ لا أعرف كيف أضيف المتغيرات
   □ لا أجد زر "Deploy"
   □ خطوة أخرى (اذكرها)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📸 4. (اختياري) أرسل لقطة شاشة

   إذا أمكن، أرسل لقطة شاشة أو صف ما تراه

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🚀 بعد إخباري بالتفاصيل، سأساعدك خطوة بخطوة!
