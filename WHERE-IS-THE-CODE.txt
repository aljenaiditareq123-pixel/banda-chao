   🔍 أين الكود؟ - Where is the Code?
═══════════════════════════════════════════════════════

📋 الكود موجود في Terminal!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🎯 الخطوات:

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1️⃣ افتح Terminal (الذي تشغّل فيه vercel login)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

2️⃣ ابحث في Terminal عن هذا النص:

   Visit https://vercel.com/oauth/device?user_code=XXXX-XXXX
   
   أو
   
   user_code=XXXX-XXXX

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

3️⃣ الكود هو الجزء بعد "user_code="

   مثال:
   user_code=DVZT-XXXT
   
   → الكود هو: DVZT-XXXT

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

4️⃣ إذا لم تجد الكود:

   → قد يكون الكود انتهت صلاحيته
   → نفّذ الأمر مرة أخرى:
   
   npx vercel login
   
   → سيظهر رابط وكود جديد

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

💡 ما هي "الحقول" في الصفحة؟

   → الصفحة تحتوي على 8 مربعات صغيرة
   → كل مربع لحرف واحد
   → أدخل حرفاً واحداً في كل مربع

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📸 إذا سجلت الشاشة:

   → أرسلها لي وسأساعدك تحديداً!
   → أو اشرح لي ماذا ترى في Terminal

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🚀 ابدأ: افتح Terminal وابحث عن "user_code="
