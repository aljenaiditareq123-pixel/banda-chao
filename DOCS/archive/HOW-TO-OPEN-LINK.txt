   ✅ كيف تفتح الرابط في المتصفح - How to Open Link
═══════════════════════════════════════════════════════

❌ خطأ شائع: لا تضع الرابط في Terminal!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

✅ الطريقة الصحيحة:

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 1: انسخ الرابط من Terminal

   → ابحث في Terminal عن:
      https://vercel.com/oauth/device?user_code=DVZT-XXXT
   
   → حدد الرابط بالماوس
   → اضغط Cmd + C (للنسخ)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 2: افتح المتصفح (Safari أو Chrome)

   → اضغط Cmd + T (لفتح تبويب جديد)
   → أو افتح Safari/Chrome

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 3: الصق الرابط في شريط العنوان

   → اضغط Cmd + V (للصق)
   → اضغط Enter

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📋 الخطوة 4: اضغط "Authorize"

   → في صفحة Vercel
   → اضغط الزر الأزرق "Authorize"
   → ✅ سيرجع Terminal تلقائياً!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

💡 ملاحظة مهمة:

   ❌ لا تضع الرابط في Terminal!
   ✅ افتحه في المتصفح (Safari أو Chrome)!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

🚀 الخطوات بسرعة:

   1. انسخ الرابط من Terminal (Cmd + C)
   2. افتح Safari/Chrome
   3. الصق في شريط العنوان (Cmd + V)
   4. اضغط Enter
   5. اضغط "Authorize"
   6. ✅ تم!
